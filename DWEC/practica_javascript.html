JS Loop While 
 <!DOCTYPE html>
<html>
<body>

<p>Ejemplo de un for que muestra numeros hasta el que yo quiera.</p>

<button onclick="myFunction()">Click aqui</button>

<p id="demo"></p>

<script>
function myFunction() {
    var text = "";
    var i = 0;
    while (i < 10) {
        text += "<br>Este numero es el " + i;
        i++;
    }
    document.getElementById("demo").innerHTML = text;
}
</script>
